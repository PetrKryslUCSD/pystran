 93def beam_2d_bending_stiffness(e_z, h, E, I_y):
 94    """
 95    Compute 2d beam stiffness matrix.
 96
 97    Two-point Gauss quadrature is used to compute the stiffness matrix.
 98
 99    The formula reads
100
101    $K = (h/2) \\int_{-1}^{+1} EI_y B^T B  d\\xi$,
102
103    where $B$ is the curvature-displacement matrix (computed by
104    `beam_2d_curv_displ_matrix`), and $h/2$ is the Jacobian. `I_y` is the second
105    moment of area about the `y` axis (which is orthogonal to the plane of
106    bending).
107    """
108    xiG, WG = gauss.rule(2)
109    K = zeros((6, 6))
110    for xi, W in zip(xiG, WG):
111        B = beam_2d_curv_displ_matrix(e_z, h, xi)
112        K += E * I_y * outer(B.T, B) * W * (h / 2)
113    return K

116def beam_3d_xz_curv_displ_matrix(e_y, e_z, h, xi):
117    """
118    Compute beam curvature-displacement matrix in the local `x-z` plane (i.e.
119    bending about the `y` axis).
120
121    The curvature $d^2w/dx^2$ is computed in the local coordinate system of the
122    beam as $d^2w/dx^2 = B U$. Here $B$ is the curvature-displacement matrix and
123    $U$ is the displacement vector. All three displacement components and all
124    three rotation components at each joint are assumed, so the matrix $B$ has
125    one row and twelve columns.
126    """
127    d2Ndxi2 = beam_3d_xz_shape_fun_xi2(xi)
128    B = zeros((1, 12))
129    B[0, 0:3] = d2Ndxi2[0] * (2 / h) ** 2 * e_z
130    B[0, 3:6] = (h / 2) * d2Ndxi2[1] * (2 / h) ** 2 * e_y
131    B[0, 6:9] = d2Ndxi2[2] * (2 / h) ** 2 * e_z
132    B[0, 9:12] = (h / 2) * d2Ndxi2[3] * (2 / h) ** 2 * e_y
133    return B

156def beam_3d_bending_stiffness(e_y, e_z, h, E, Iy, Iz):
157    """
158    Compute 3d beam stiffness matrices for bending in the planes `x-y` and
159    `x-z`.
160
161    Two-point Gauss quadrature is used to compute the stiffness matrices.
162
163    The formula reads
164
165    $K = (h/2) \\int_{-1}^{+1} EI_y B_{xz}^T B_{xz}  d\\xi$,
166
167    for bending in the `x-z` plane, and
168
169    $K = (h/2) \\int_{-1}^{+1} EI_z B_{xy}^T B_{xy}  d\\xi$,
170
171    where $B_{xz}$ is the curvature-displacement matrix for bending in the `x-z`
172    plane (computed by `beam_3d_xz_curv_displ_matrix`), $B_{xy}$ is the
173    curvature-displacement matrix for bending in the `x-y` plane (computed by
174    `beam_3d_xy_curv_displ_matrix`), and $h/2$ is the Jacobian. `I_y` is the
175    second moment of area about the `y` axis, and `I_z`  is the second moment of
176    area about the `z` axis.
177    """
178    xiG, WG = gauss.rule(2)
179    Kxy = zeros((12, 12))
180    for xi, W in zip(xiG, WG):
181        B = beam_3d_xy_curv_displ_matrix(e_y, e_z, h, xi)
182        Kxy += E * Iz * outer(B.T, B) * W * (h / 2)
183    Kxz = zeros((12, 12))
184    for xi, W in zip(xiG, WG):
185        B = beam_3d_xz_curv_displ_matrix(e_y, e_z, h, xi)
186        Kxz += E * Iy * outer(B.T, B) * W * (h / 2)
187    return Kxy, Kxz

190def beam_2d_curv_displ_matrix(e_z, h, xi):
191    """
192    Compute 2d beam curvature-displacement matrix.
193
194    Here the curvatures is with respect to the physical coordinate measured
195    along the member (local `x`).
196
197    The curvature $d^2w/dx^2$ is computed in the local coordinate system of the
198    beam as $d^2w/dx^2 = B U$. Here $B$ is the curvature-displacement matrix and
199    $U$ is the displacement vector. Two displacement components and one rotation
200    component at each joint are assumed, so the matrix $B$ has one row and six
201    columns.
202    """
203    d2Ndxi2 = herm_basis_xi2(xi)
204    B = zeros((1, 6))
205    B[0, 0:2] = d2Ndxi2[0] * (2 / h) ** 2 * e_z
206    B[0, 2] = (h / 2) * d2Ndxi2[1] * (2 / h) ** 2
207    B[0, 3:5] = d2Ndxi2[2] * (2 / h) ** 2 * e_z
208    B[0, 5] = (h / 2) * d2Ndxi2[3] * (2 / h) ** 2
209    return B

280def beam_2d_moment(member, i, j, xi):
281    """
282    Compute 2d beam moment based on the displacements stored at the joints.
283    The moment is computed at the parametric location `xi` along the beam.
284
285    The moment is mathematically defined as $M = -EI d^2w/dx^2$.
286
287    The curvature is computed with the curvature-displacement matrix $B$ by the function
288    `beam_2d_curv_displ_matrix`.
289    """
290    e_x, e_z, h = geometry.member_2d_geometry(i, j)
291    sect = member["section"]
292    E, I = sect["E"], sect["I"]
293    ui, uj = i["displacements"], j["displacements"]
294    u = concatenate([ui, uj])
295    B = beam_2d_curv_displ_matrix(e_z, h, xi)
296    return -E * I * dot(B, u)

299def beam_3d_moment(member, i, j, axis, xi):
300    """
301    Compute 3d beam moment based on the displacements stored at the joints. The
302    moment is computed at the parametric location `xi` along the beam. The
303    moment acts about the axis specified by the string `axis` (`'y'` or `'z'`).
304
305    The moments are mathematically defined as $M_y = -EI_y d^2w/dx^2$ for
306    bending about the `y` axis, and $M_z = +EI_z d^2v/dx^2$ for bending about
307    the `z` axis.
308
309    The curvatures are computed with  curvature-displacement matrices $B$ by the
310    functions `beam_3d_xz_curv_displ_matrix` and `beam_3d_xy_curv_displ_matrix`,
311    respectively.
312    """
313    sect = member["section"]
314    e_x, e_y, e_z, h = geometry.member_3d_geometry(i, j, sect["xz_vector"])
315    E, Iy, Iz = sect["E"], sect["Iy"], sect["Iz"]
316    ui, uj = i["displacements"], j["displacements"]
317    u = concatenate([ui, uj])
318    if axis == "y":
319        B = beam_3d_xz_curv_displ_matrix(e_y, e_z, h, xi)
320        M = -E * Iy * dot(B, u)
321    else:
322        B = beam_3d_xy_curv_displ_matrix(e_y, e_z, h, xi)
323        M = +E * Iz * dot(B, u)
324    return M

327def beam_3d_torsion_moment(member, i, j):
328    """
329    Compute 3d beam torsion moment based on the displacements stored at the
330    joints. The moment is uniform along the beam.
331
332    The moment is mathematically defined as $T = GJ d\\theta_x/dx$. The rate of
333    change of the axial rotation, $d\\theta_x/dx$, is computed with the
334    torsion-displacement matrix $B$, obtained by the function
335    `beam_3d_torsion_displ_matrix`.
336
337    The torsion moment is uniform along the beam.
338    """
339    sect = member["section"]
340    e_x, e_y, e_z, h = geometry.member_3d_geometry(i, j, sect["xz_vector"])
341    G, J = sect["G"], sect["J"]
342    ui, uj = i["displacements"][3:6], j["displacements"][3:6]
343    u = concatenate([ui, uj])
344    B = beam_3d_torsion_displ_matrix(e_x, h)
345    T = G * J * dot(B, u)
346    return T

464def assemble_stiffness(Kg, member, i, j):
465    """
466    Assemble beam stiffness matrix.
467
468    Beam is here considered as a superposition of four mechanisms  -- axial bar,
469    torsion bar, bending in the `x-y` plane, and bending in the `x-z` plane.
470    """
471    # Add stiffness in bending.
472    beam_is_2d = len(i["coordinates"]) == len(j["coordinates"]) == 2
473    dof = concatenate([i["dof"], j["dof"]])
474    if beam_is_2d:
475        sect = member["section"]
476        e_x, e_z, h = geometry.member_2d_geometry(i, j)
477        # Add stiffness in bending.
478        E, I = sect["E"], sect["I"]
479        k = beam_2d_bending_stiffness(e_z, h, E, I)
480        Kg = assemble.assemble(Kg, dof, k)
481        # Add stiffness in the axial direction.
482        E, A = sect["E"], sect["A"]
483        k = beam_2d_3d_axial_stiffness(e_x, h, E, A)
484        dof = concatenate([i["dof"][0:2], j["dof"][0:2]])
485        Kg = assemble.assemble(Kg, dof, k)
486    else:
487        sect = member["section"]
488        e_x, e_y, e_z, h = geometry.member_3d_geometry(i, j, sect["xz_vector"])
489        # Add stiffness in bending.
490        E, Iy, Iz = sect["E"], sect["Iy"], sect["Iz"]
491        kxy, kxz = beam_3d_bending_stiffness(e_y, e_z, h, E, Iy, Iz)
492        Kg = assemble.assemble(Kg, dof, kxy)
493        Kg = assemble.assemble(Kg, dof, kxz)
494        # Add stiffness in the axial direction.

528    """
529    Compute beam mass matrix.
530
531    The mass matrix is consistent, which means that it is computed as discrete
532    form of the kinetic energy of the element,
533
534    $\\int \\rho A \\left(\\dot u \\cdot \\dot u +  \\dot w \\cdot \\dot
535    w\\right)dx$
536
537    where $\\dot u$ and $\\dot w$ are the velocities in the $x$ and $z$
538    directions.
539
540    The velocity $\\dot u$ is assumed to very linearly along the element, and
541    the velocity $\\dot w$ is assumed to vary according to the Hermite shape
542    functions.
543
544    Gauss quadrature is used to compute the mass matrix.
545    """
546    xiG, WG = gauss.rule(4)
547    n = (len(e_x) + 1) * 2
548    m = zeros((n, n))
549    for xi, W in zip(xiG, WG):
550        N = geometry.lin_basis(xi)
551        extN = concatenate([N[0] * e_x, [0.0], N[1] * e_x, [0.0]])
552        m += rho * A * outer(extN, extN) * W * (h / 2)
553        N = geometry.herm_basis(xi)
554        extN = concatenate([N[0] * e_z, [(h / 2) * N[1]], N[2] * e_z, [(h / 2) * N[3]]])

559def beam_3d_mass(e_x, e_y, e_z, h, rho, A, Ix, Iy, Iz):
560    """
561    Compute beam mass matrix.
562
563    The mass matrix is consistent, which means that it is computed as discrete
564    form of the kinetic energy of the element,
565
566    $\\int \\rho A \\left(\\dot u \\cdot \\dot u + \\dot v \\cdot \\dot v + \\dot w \\cdot \\dot
567    w\\right)dx$
568
569    where $\\dot u$, $\\dot v$, and $\\dot w$ are the velocities in the $x$, $y$, and $z$
570    directions.
571
572    The velocity $\\dot u$ is assumed to very linearly along the element, and
573    the velocity $\\dot v$, $\\dot w$ is assumed to vary according to the Hermite shape
574    functions. Rotations about $y$ and $z$ are ignored in the kinetic energy.
575
576    For spinning of the beam about its axis, the kinetic energy is given by the formula
577
578    $\\int \\rho Ix \\dot \\theta_x \\cdot \\dot \\theta_x dx$.
579
580    Gauss quadrature is used to compute the mass matrix.
581    """
582    xiG, WG = gauss.rule(4)
583    n = len(e_x) * 4
584    m = zeros((n, n))
585    for xi, W in zip(xiG, WG):
586        N = geometry.lin_basis(xi)
587        # Translations
588        extN = concatenate([N[0] * e_x, [0.0, 0.0, 0.0], N[1] * e_x, [0.0, 0.0, 0.0]])
589        m += rho * A * outer(extN, extN) * W * (h / 2)
590        extN = concatenate([N[0] * e_y, [0.0, 0.0, 0.0], N[1] * e_y, [0.0, 0.0, 0.0]])
591        m += rho * A * outer(extN, extN) * W * (h / 2)
592        extN = concatenate([N[0] * e_z, [0.0, 0.0, 0.0], N[1] * e_z, [0.0, 0.0, 0.0]])
593        m += rho * A * outer(extN, extN) * W * (h / 2)
594        # Torsion
595        extN = concatenate([[0.0, 0.0, 0.0], N[0] * e_x, [0.0, 0.0, 0.0], N[1] * e_x])
596        m += rho * Ix * outer(extN, extN) * W * (h / 2)
597        # Transverse displacements and rotations about y and z
598        N = beam_3d_xz_shape_fun(xi)
599        extN = concatenate(
600            [N[0] * e_z, (h / 2) * N[1] * e_y, N[2] * e_z, (h / 2) * N[3] * e_y]
601        )
602        m += rho * Iy * outer(extN, extN) * W * (h / 2)
603        N = beam_3d_xy_shape_fun(xi)
604        extN = concatenate(
605            [N[0] * e_y, (h / 2) * N[1] * e_z, N[2] * e_y, (h / 2) * N[3] * e_z]
606        )
607        m += rho * Iz * outer(extN, extN) * W * (h / 2)
608
609    return m
